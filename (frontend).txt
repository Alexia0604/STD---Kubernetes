// app/[slug]/page.tsx
import { headers as getHeaders } from 'next/headers'
import { notFound } from 'next/navigation'
import { getPayload } from 'payload'
import React from 'react'
import config from '../../payload.config'

// Acesta e tipul corect aÈ™teptat de Next.js pentru `params`:
type Props = {
  params: {
    slug: string
  }
}

export default async function DynamicPage({ params }: Props) {
  const headers = getHeaders()
  const payloadInstance = await getPayload({ config })

  const pageSlug = params.slug || 'home'

  const pagesResponse = await payloadInstance.find({
    collection: 'pages',
    where: {
      slug: {
        equals: pageSlug,
      },
    },
  })

  const page = pagesResponse.docs[0]

  if (!page) {
    return notFound()
  }

  const showChat = page?.showChat || false
  const showIA = page?.showIA || false

  return (
    <div className="dynamic-page">
      <div className="page-header">
        <h1>{page.title}</h1>
      </div>

      {page.content && (
        <div
          className="page-content content-wrapper"
          dangerouslySetInnerHTML={{ __html: page.content }}
        />
      )}

      {showChat && (
        <div className="chat-container">
          <div className="chat-header">
            <h2>Travel Community Chat</h2>
            <p>Connect with travelers worldwide</p>
          </div>
          <div className="chat-frame">
            <iframe
              src="http://localhost:30090"
              width="100%"
              height="600px"
              frameBorder="0"
              title="Chat Application"
            />
          </div>
        </div>
      )}

      {showIA && (
        <div className="ai-container">
          <div className="ai-header">
            <h2>Travel Photo AI</h2>
            <p>Organize your travel memories</p>
          </div>
          <div className="ai-frame">
            <iframe
              src="http://localhost:3001"
              width="100%"
              height="600px"
              frameBorder="0"
              title="AI Application"
            />
          </div>
        </div>
      )}

      <div className="related-content">
        <h2>You might also like</h2>
        <div className="content-grid">
          <div className="content-card">
            <img src="/api/placeholder/300/200" alt="Related content" />
            <h3>Top 10 Beach Destinations</h3>
            <p>Discover the most beautiful beaches around the world</p>
          </div>
          <div className="content-card">
            <img src="/api/placeholder/300/200" alt="Related content" />
            <h3>Budget Travel Tips</h3>
            <p>How to travel more while spending less</p>
          </div>
        </div>
      </div>
    </div>
  )
}
